const LOCATION_CACHE_KEY = "abgains_location";
const LOCATION_TIME_KEY  = "abgains_location_time";
const CACHE_TIME         = 30 * 60 * 1000; // 30 min
const MOVE_KM            = 0.5;

/* ==========================
   DISTANCE CALC
========================== */
function distanceKm(lat1, lon1, lat2, lon2) {
    const R = 6371;
    const dLat = (lat2 - lat1) * Math.PI / 180;
    const dLon = (lon2 - lon1) * Math.PI / 180;
    const a =
        Math.sin(dLat / 2) ** 2 +
        Math.cos(lat1 * Math.PI / 180) *
        Math.cos(lat2 * Math.PI / 180) *
        Math.sin(dLon / 2) ** 2;
    return 2 * R * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
}

/* ==========================
   SMART LOCATION (AUTO)
========================== */
function getLocationSmart(force = false) {
    const cached = localStorage.getItem(LOCATION_CACHE_KEY);
    const cachedTime = localStorage.getItem(LOCATION_TIME_KEY);

    if (!force && cached && cachedTime) {
        const age = Date.now() - Number(cachedTime);
        if (age < CACHE_TIME) {
            const loc = JSON.parse(cached);
            fetchCityName(loc.lat, loc.lng);
            return;
        }
    }

    if (!navigator.geolocation) {
        showLocationError("Geolocation not supported");
        return;
    }

    navigator.geolocation.getCurrentPosition(
        pos => {
            const lat = pos.coords.latitude;
            const lng = pos.coords.longitude;

            if (!force && cached) {
                const old = JSON.parse(cached);
                if (distanceKm(old.lat, old.lng, lat, lng) < MOVE_KM) {
                    fetchCityName(old.lat, old.lng);
                    return;
                }
            }

            localStorage.setItem(LOCATION_CACHE_KEY, JSON.stringify({ lat, lng }));
            localStorage.setItem(LOCATION_TIME_KEY, Date.now().toString());

            fetchCityName(lat, lng);
        },
        () => showLocationError("Location permission denied"),
        { enableHighAccuracy: false, maximumAge: 600000, timeout: 5000 }
    );
}

/* ==========================
   MANUAL BUTTON
========================== */
function forceUpdateLocation() {
    localStorage.removeItem(LOCATION_CACHE_KEY);
    localStorage.removeItem(LOCATION_TIME_KEY);
    getLocationSmart(true);
}
